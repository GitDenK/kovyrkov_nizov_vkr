\begin{description}
    \setlength{\itemsep}{2pt}
    \item[API] \textit{(Application Programming Interface)} --- программный интерфейс приложения; набор соглашений о формате вызовов и обмене данными между программными компонентами.
    \item[ЕГЭ] Единый государственный экзамен.
    \item[JSON] \textit{(JavaScript Object Notation)} --- текстовый формат обмена структурированными данными на основе синтаксиса JavaScript.
    \item[LLM] \textit{(Large Language Model)} --- большая языковая модель; нейросетевая архитектура, обученная на текстовых корпусах для генерации и понимания текста.
    \item[RAG] \textit{(Retrieval-Augmented Generation)} --- генерация с извлечением информации; техника дополнения запроса к языковой модели релевантными фрагментами из корпуса.
    \item[SVG] \textit{(Scalable Vector Graphics)} --- формат масштабируемой векторной графики на основе XML, стандартизированный консорциумом W3C.
    \item[ФИПИ] Федеральный институт педагогических измерений.
    \item[HTML] \textit{(HyperText Markup Language)} --- язык гипертекстовой разметки; стандарт описания структуры веб-страниц.
    \item[URL] \textit{(Uniform Resource Locator)} --- унифицированный указатель ресурса; стандартный способ адресации ресурсов в сети Интернет.
\end{description}

\subsection*{В.2. Pydantic-схема сцены}

Файл \texttt{svg-generator/SVG\_generator/service/app/schemas/scene.py},
строки~86--103.
Класс \texttt{Scene} является центральным контрактом системы: именно в этом
формате конвертер передаёт описание сцены рендереру.
Поле \texttt{constraints} содержит декларативные геометрические ограничения

\subsection*{В.3. Решатель геометрических ограничений}
% ───────────────────────────────────────────────────────────

Файл \texttt{svg-generator/SVG\_generator/service/app/solver/solver.py},
строки~6--28 и 60--116 (инициализация + случаи высоты и биссектрисы;
случаи \texttt{midpoint} и \texttt{median} устроены аналогично и опущены).
Функция \texttt{solve} разворачивает декларативные ограничения сцены в конкретные
объекты (точки и отрезки), вычисляя координаты программно.
Тем самым обеспечивается точность геометрических построений независимо от
языковой модели: координаты высоты и биссектрисы треугольника получаются по
формулам, а не из ответа LLM.

\begin{lstlisting}[caption={solver.py, строки 6--28 и 60--116: инициализация и вычисление высоты и биссектрисы (случаи midpoint и median аналогичны и опущены).}]
def solve(scene: Scene) -> tuple[Scene, list[str]]:
    """Обрабатывает constraints, добавляет вычисленные объекты в scene."""
    warnings: list[str] = []
    new_objs: list = list(scene.objects)

    # Индексы объектов по id (обновляются при добавлении новых)
    pts: dict[str, Point] = {}
    objs: dict = {}
    for obj in new_objs:
        objs[obj.id] = obj
        if obj.type == "point":
            pts[obj.id] = obj

    existing_ids = set(objs.keys())

    def _add(obj) -> None:
        if obj.id not in existing_ids:
            new_objs.append(obj)
            objs[obj.id] = obj
            existing_ids.add(obj.id)
            if obj.type == "point":
                pts[obj.id] = obj

    for c in scene.constraints:
        # ... (midpoint, median) ...

        elif c.type == "altitude":
            tri = objs.get(c.triangle)
            if not tri:
                warnings.append(f"Constraint '{c.id}': triangle '{c.triangle}' not found")
                continue
            if c.vertex not in tri.vertices:
                warnings.append(f"Constraint '{c.id}': vertex '{c.vertex}' not in triangle")
                continue
            others = [v for v in tri.vertices if v != c.vertex]
            pv = pts.get(c.vertex)
            p1, p2 = pts.get(others[0]), pts.get(others[1])
            if not all([pv, p1, p2]):
                warnings.append(f"Constraint '{c.id}': points not found")
                continue
            # Основание перпендикуляра из pv на отрезок p1-p2
            dx, dy = p2.x - p1.x, p2.y - p1.y
            denom = dx * dx + dy * dy
            if denom == 0:
                warnings.append(f"Constraint '{c.id}': degenerate altitude base")
                continue
            t = ((pv.x - p1.x) * dx + (pv.y - p1.y) * dy) / denom
            foot_id = f"{c.id}_foot"
            seg_id = f"{c.id}_seg"
            _add(Point(id=foot_id, x=p1.x + t * dx, y=p1.y + t * dy))
            _add(Segment(id=seg_id, from_point=c.vertex, to_point=foot_id))

        elif c.type == "bisector":
            # ...проверки triangle / vertex / points аналогичны altitude...
            others = [v for v in tri.vertices if v != c.vertex]
            pv = pts.get(c.vertex)
            p1, p2 = pts.get(others[0]), pts.get(others[1])
            # Теорема о биссектрисе: D делит p1-p2 в отношении |pv-p1|:|pv-p2|
            d1 = math.dist((pv.x, pv.y), (p1.x, p1.y))
            d2 = math.dist((pv.x, pv.y), (p2.x, p2.y))
            total = d1 + d2
            if total == 0:
                warnings.append(f"Constraint '{c.id}': degenerate bisector")
                continue
            bx = (d2 * p1.x + d1 * p2.x) / total
            by = (d2 * p1.y + d1 * p2.y) / total
            bis_id = f"{c.id}_pt"
            seg_id = f"{c.id}_seg"
            _add(Point(id=bis_id, x=bx, y=by))
            _add(Segment(id=seg_id, from_point=c.vertex, to_point=bis_id))

        # right_angle_marker: пропускается, не добавляет объекты

    return scene.model_copy(update={"objects": new_objs}), warnings
\end{lstlisting}
